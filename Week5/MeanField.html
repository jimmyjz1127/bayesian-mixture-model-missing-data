Assume Sigma is known, and estimate $\pi$ by MLE

We want to compute the intractable GMM posterior: $$ p(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi}|\mathbf{X}) = \frac{p(\mathbf{X},\boldsymbol{\mu},\mathbf{z},\boldsymbol{\pi})}{p(\mathbf{X})} $$

Introduce approximating distribution $q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})$ where we use KL divergence as a cost function $$ \mathbb{KL}(q||p) = \sum_{\mathbf{z},\boldsymbol{\mu}}q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})\log\frac{q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})}{p(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi}|\mathbf{X})} $$

Re-write as objective function $$\begin{align} \mathcal{J}(q) &= \mathbb{KL}(q||p) = \int\sum_{\mathbf{z}}q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})\log\frac{q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})}{p(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi}|\mathbf{X})}d\boldsymbol{\mu}\\ &= \int\sum_{\mathbf{z}}q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})\log q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})d\boldsymbol{\mu} - \int\sum_{\mathbf{z}}q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})\log p(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi}|\mathbf{X})d\boldsymbol{\mu}\\ &= \int\sum_{\mathbf{z}}q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})\log q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})d\boldsymbol{\mu} -\int\sum_{\mathbf{z}}q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})\log \frac{p(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi},\mathbf{X})}{p(\mathbf{X})}d\boldsymbol{\mu}\\ &= \int\sum_{\mathbf{z}}q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})\log q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})d\boldsymbol{\mu} - \int\sum_{\mathbf{z}}q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})\log p(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi},\mathbf{X})d\boldsymbol{\mu} + \int\sum_{\mathbf{z}}q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})\log p(\mathbf{X})d\boldsymbol{\mu}\\ &= \int\sum_{\mathbf{z}}q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})\log q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})d\boldsymbol{\mu} - \int\sum_{\mathbf{z}}q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})\log p(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi},\mathbf{X})d\boldsymbol{\mu} + p(\mathbf{X})\\ &= \int\sum_{\mathbf{z}}q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})\log \frac{q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})}{p(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi},\mathbf{X})}d\boldsymbol{\mu} + p(\mathbf{X})\\ &= \mathbb{KL}\Big(q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi}) || p(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi},\mathbf{X})\Big) + p(\mathbf{X}) \end{align}$$

Note $$ \mathcal{J}(q) = \mathbb{KL}\Big(q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi}) || p(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi},\mathbf{X})\Big) + \underbrace{p(\mathbf{X})}_{\text{constant}} $$

Hence our objective can simplify to $$\begin{align} q* &\leftarrow \argmin_{q} \left[ \mathbb{KL}\Big(q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi}) || p(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi},\mathbf{X})\Big) + \underbrace{p(\mathbf{X})}_{\text{constant}} \right]\\ &= \argmin_{q} \left[ \mathbb{KL}\Big(q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi}) || p(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi},\mathbf{X})\Big) \right] \end{align}$$

Define new objective function as $\mathcal{L}(q) = -\mathcal{J}(q) = -\mathbb{KL}\Big(q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi}) || p(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi},\mathbf{X})\Big) - p(\mathbf{X})$

We can show that $\mathcal{L}(q)$ is ELBO for data log likelihood : $$\begin{align} \log p(\mathbf{X}) = \log \int \sum_z p(\mathbf{X},\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})d\mu \end{align}$$ The following always holds $$\begin{align} \log p(\mathbf{X}) = \log \int \sum_z q(\boldsymbol{\mu},\mathbf{z},\boldsymbol{\pi}) \cdot \frac{p(\mathbf{X},\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})}{q(\boldsymbol{\mu},\mathbf{z},\boldsymbol{\pi})} d\mu \end{align}$$
Jensen's Inequality gives the following $$\begin{align} &\log p(\mathbf{X}) = \log \int \sum_z q(\boldsymbol{\mu},\mathbf{z},\boldsymbol{\pi}) \cdot \frac{p(\mathbf{X},\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})}{q(\boldsymbol{\mu},\mathbf{z},\boldsymbol{\pi})} d\mu \geq \int \sum_z q(\boldsymbol{\mu},\mathbf{z},\boldsymbol{\pi}) \cdot \log \frac{p(\mathbf{X},\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})}{q(\boldsymbol{\mu},\mathbf{z},\boldsymbol{\pi})} d\mu\\\\ &\implies \log p(\mathbf{X}) \geq \int \sum_z q(\boldsymbol{\mu},\mathbf{z},\boldsymbol{\pi}) \cdot \log \frac{p(\mathbf{X},\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})}{q(\boldsymbol{\mu},\mathbf{z},\boldsymbol{\pi})} d\mu\\\\ &\implies \log p(\mathbf{X}) \geq -\mathbb{KL}\Big(q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi}) || p(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi},\mathbf{X})\Big)\\\\ &\implies \log p(\mathbf{X}) \geq \mathcal{L}(q) \end{align}$$ Therefore by maximizing $\mathcal{L}$, we bring $q$ closer to true posterior and maximize ELBO of marginal data log likelihood.

We have $$ \mathcal{L}(q) = \mathbb{E}_{q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})}[\log p(\mathbf{X},\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})] - \mathbb{E}_{q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})}[\log q(\mathbf{z},\boldsymbol{\mu},\boldsymbol{\pi})] $$

And want to compute $$ \frac{d\mathcal{L}}{q(\boldsymbol{\mu},\mathbf{z},\boldsymbol{\pi})} = 0 $$

Mean Field Independence Simplifying Assumption : $$ q(\boldsymbol{\mu},\mathbf{z},\boldsymbol{\pi}) = q(\boldsymbol{\pi})\prod_n^N \underbrace{q_n(z_n)}_{\text{Cat}} \prod_k^K \underbrace{q_k(\mu_k)}_{\text{Gaussian}} $$

  4. Sigma is known

Updates for variational posterior of model parameters $q(\boldsymbol{\pi},\boldsymbol{\mu})$

$$\begin{align} \log q(\boldsymbol{\pi},\boldsymbol{\mu}) &= \mathbb{E}_{q(\mathbf{z})}\Big[\log p(\mathbf{X},\boldsymbol{\mu},\mathbf{z})\Big] + \text{const}\\ &= \mathbb{E}_{q(\mathbf{z})}\Big[\log p(\mathbf{X}|\boldsymbol{\mu},\mathbf{z})p(\boldsymbol{\mu})p(\mathbf{z}|\boldsymbol{\pi})p(\boldsymbol{\pi})\Big] + \text{const}\\ &= \mathbb{E}_{q(\mathbf{z})}\Big[ \log {p(\boldsymbol{\pi})} + \log p(\mathbf{z}|\boldsymbol{\pi}) + \sum_n^N \log p(\mathbf{x}_n|\boldsymbol{\mu}_{z_n},z_n) \Big] + \log p(\boldsymbol{\mu}) + \text{const}\\ &= \log p(\boldsymbol{\pi}) + \mathbb{E}_{q(\mathbf{z})}\Big[ \log p(\mathbf{z}|\boldsymbol{\pi}) \Big] + \sum_n^N \mathbb{E}_{q(z_n)}\Big[ \log p(\mathbf{x}_n|\boldsymbol{\mu}_{z_n},z_n) \Big] + + \log p(\boldsymbol{\mu}) + \text{const}\\ \end{align}$$

Can separate RHS into terms containing $\pi$ and $\mu$ for update steps $q(\pi)$ and $q(\mu)$

$$\begin{align} \log q(\boldsymbol{\pi}) &= \log p(\boldsymbol{\pi}) + \mathbb{E}_{q(\mathbf{z})}\Big[ \log p(\mathbf{z}|\boldsymbol{\pi}) \Big] + \text{const}\\ &= \log \text{Dir}(\boldsymbol{\pi}|\boldsymbol{\alpha}_0) + \mathbb{E}_{q(\mathbf{z})}\Big[ \sum_n^N \log p(z_n|\boldsymbol{\pi}) \Big] + \text{const}\\ &= \log \Big[ \frac{1}{B(\alpha_0)}\prod_k^K \pi_k^{\alpha_0 - 1} \Big] + \sum_n^N \mathbb{E}_{q(z_n)} \Big[\log p(z_n|\boldsymbol{\pi}) \Big] + \text{const}\\ &= \underbrace{-\log B(\alpha_0)}_{\text{Independent of $\pi$}} + \sum_k^K \log \pi_k^{\alpha_0 - 1} + \sum_n^N \mathbb{E}_{q(z_n)} \Big[\log \pi_{z_n} \Big] + \text{const}\\ &= \sum_k^K (\alpha_0 - 1)\log \pi_k + \sum_n^N \sum_k^K q(z_n = k) \cdot \log \pi_{k} + \text{const}, \quad -\log B(\alpha_0) \text{ absorbed into const}\\ &= (\alpha_0 - 1) \sum_k^K \log \pi_k + \sum_n^N \sum_k^K r_{nk} \cdot \log \pi_{k} + \text{const}\\ \implies & \boxed{q(\boldsymbol{\pi}) = \text{Dir}(\boldsymbol{\pi}|\boldsymbol{\alpha}_0 + N)} \end{align}$$ And since $q(\boldsymbol{\mu}) = \prod_k^K q_k(\mu_k)$ $$\begin{align} \log q(\boldsymbol{\mu}) &= \sum_n^N \mathbb{E}_{q(z_n)}\Big[ \log p(\mathbf{x}_n|\boldsymbol{\mu}_{z_n},z_n) \Big] + \log p(\boldsymbol{\mu}) +\text{const}\\ &= \sum_n^N \sum_k^K \Big[ q(z_n = k) \log p(\mathbf{x}_n|\boldsymbol{\mu}_{k},z_n=k) \Big] + \sum_k^K \log p(\mu_k) +\text{const}\\ &= \sum_n^N \sum_k^K \Big[ r_{nk} \cdot \log p(\mathbf{x}_n|\boldsymbol{\mu}_{k},z_n=k) \Big] + \sum_k^K \log p(\mu_k) + \text{const}\\ &= \sum_n^N \sum_k^K \Big[ r_{nk} \cdot \log \mathcal{N}(\mathbf{x}_n|\boldsymbol{\mu}_{k},\Sigma_k) \Big] + \sum_k^K \log \mathcal{N}(\mu_k|\mathbf{m}_{0,k},\mathbf{V}_{0,k}) + \text{const}\\\\ \implies \log q_{\mu_k}(\mu_k) &= \log \mathcal{N}(\mu_k|\mathbf{m}_{0,k}, \mathbf{V}_{0,k}) + \sum_n^N \gamma_{n,k} \cdot \log \mathcal{N}(\mathbf{x}_n|\mu_k,\Sigma_k) + \text{const}\\ \end{align}$$

Leverage conjugacy (Murphy 128) to derive $q(\mu_k)$. With this we get valid normalized distribution and can ignore constant term. $$\begin{align} q_{\mu_k}(\mu_k) &= \mathcal{N}\Big(\mu_k|\mathbf{m}_k,\mathbf{V}_k \Big)\\ & N_k = \sum_n^N \gamma_{n,k}\\ & \bar{\mathbf{x}}_k = \frac{1}{N_k}\sum_n^N \gamma_{n,k} \cdot \mathbf{x}_n\\ & \mathbf{V}_k = \Big(\mathbf{V}_{0,k}^{-1} + N_k\Sigma_k^{-1}\Big)^{-1}\\ & \mathbf{m}_k = \mathbf{V}_k \Big(\mathbf{V}_{0,k}^{-1}\mathbf{m}_{0,k} + \Sigma_k^{-1}(N_k\bar{\mathbf{x}}_k) \Big) \end{align}$$

def update_(X, R, , _0, _0, K):
    N, D = X.shape

    S = np.zeros((K, D, D))  # Covariances of q(_k)
    m = np.zeros((K, D))     # Means of q(_k)

    nk = np.sum(R, axis=0)  # (K,)

    for k in range(K):
        _inv = np.linalg.inv([k])
        
        if nk[k] == 0:
            continue  # Avoid division by zero

        x_bar_k = np.sum(R[:, k, None] * X, axis=0) / nk[k]  # (D,)

        Sk_inv = _0 + nk[k] * _inv
        Sk = np.linalg.inv(Sk_inv)

        mk = Sk @ (_0 @ _0 + _inv @ np.sum(R[:, k, None] * X, axis=0))

        S[k] = Sk
        m[k] = mk

    return S, m

def update_z(X,m,S,,K):
    N,D = X.shape 

    R = np.zeros((N,K))

    for k in range(K):
        _inv = np.linalg.inv([k])
        diff = X - m[k]

        term1 = np.trace(_inv @ S[k])
        term2 = np.sum(diff @ _inv * diff, axis=1)

        R[:,k] = -0.5 * (term1 + term2)

    log_norm = logsumexp(R, axis=1, keepdims=True)

    R = np.exp(R - log_norm)

    loglik = np.sum(log_norm) / N

    return R,loglik

def meanfield_vi(X,K,,max_iters=100,tol=1e-4):
    N,D = X.shape

    R = rng.random((N,K))
    R /= R.sum(axis=1, keepdims=True)

    _0 = np.zeros((D)) 
    _0 = np.eye(D)

    loglikes=[]
    
    S = np.zeros((K, D, D))  # Covariances of q(_k)
    m = np.zeros((K, D))     # Means of q(_k)

    for t in range(max_iters):
        S,m = update_(X,R,,_0,_0,K)

        R,loglike = update_z(X,m,S,,K)

        loglikes.append(loglike)

        if (t > 1) and np.abs(loglikes[t] - loglikes[t-1] < tol) :
            break

    return R, S, m, loglikes
